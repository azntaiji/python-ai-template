## Task 2 - Create the markdown renderer

**Files:** `src/zach_taiji_blog/markdown_render.py`, `tests/test_markdown_render.py`

**Steps:**

1. Create `src/zach_taiji_blog/markdown_render.py` with exactly:

```python
"""Markdown-to-HTML rendering with extended syntax support."""

import markdown

_EXTENSIONS = [
    "pymdownx.mark",   # ==text== -> <mark>text</mark>
    "pymdownx.caret",  # ^text^ -> superscript
    "pymdownx.tilde",  # ~text~ -> subscript, ~~text~~ -> strikethrough
    "footnotes",
    "tables",
    "fenced_code",
    "sane_lists",
]


def render_markdown(text: str) -> str:
    """Render markdown source to an HTML fragment.

    Args:
        text: Markdown source, without YAML frontmatter.

    Returns:
        An HTML fragment as a string.
    """
    return markdown.markdown(text, extensions=_EXTENSIONS)
```

2. Create `tests/test_markdown_render.py` with exactly:

```python
"""Tests for zach_taiji_blog.markdown_render."""

from zach_taiji_blog.markdown_render import render_markdown


def test_highlight() -> None:
    assert "<mark>hot</mark>" in render_markdown("==hot==")


def test_footnote() -> None:
    html = render_markdown("body[^1]\n\n[^1]: the note")
    assert 'class="footnote"' in html


def test_table() -> None:
    html = render_markdown("| a |\n|---|\n| b |")
    assert "<table>" in html


def test_fenced_code() -> None:
    assert "<code>" in render_markdown("```\nx = 1\n```")
```
