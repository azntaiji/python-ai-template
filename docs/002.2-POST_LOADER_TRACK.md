```python
"""Load blog posts from markdown files with Obsidian-style YAML frontmatter."""

import datetime
import logging
from dataclasses import dataclass
from pathlib import Path

import frontmatter

from zach_taiji_blog.markdown_render import render_markdown

logger = logging.getLogger(__name__)


@dataclass(frozen=True)
class Post:
    """A blog post rendered to HTML, with its frontmatter metadata."""

    slug: str
    title: str
    html: str
    author: str | None = None
    date: datetime.date | None = None
    tags: tuple[str, ...] = ()


def _coerce_date(value: object) -> datetime.date | None:
    """Return a date from YAML date/datetime/ISO-string values, else None."""
    if isinstance(value, datetime.datetime):
        return value.date()
    if isinstance(value, datetime.date):
        return value
    if isinstance(value, str):
        try:
            return datetime.date.fromisoformat(value.strip())
        except ValueError:
            logger.warning("Unparseable date %r", value)
    return None


def _coerce_tags(value: object) -> tuple[str, ...]:
    """Return tags from a YAML list or comma-separated string."""
    if isinstance(value, str):
        return tuple(t.strip() for t in value.split(",") if t.strip())
    if isinstance(value, (list, tuple)):
        return tuple(str(t).strip() for t in value if str(t).strip())
    return ()


def load_post(path: Path) -> Post:
    """Parse one markdown file into a Post.

    Args:
        path: Path to a .md file; the slug is the filename without suffix.

    Returns:
        The parsed Post; title falls back to a titleized slug.
    """
    doc = frontmatter.load(path)
    meta = doc.metadata
    return Post(
        slug=path.stem,
        title=str(meta.get("title") or path.stem.replace("-", " ").title()),
        html=render_markdown(doc.content),
        author=str(meta["author"]) if meta.get("author") else None,
        date=_coerce_date(meta.get("date") or meta.get("published")),
        tags=_coerce_tags(meta.get("tags")),
    )


def load_all_posts(blogs_dir: Path) -> list[Post]:
    """Load every .md file in blogs_dir, newest first, undated posts last."""
    posts: list[Post] = []
    for path in sorted(blogs_dir.glob("*.md")):
        try:
            posts.append(load_post(path))
        except Exception:
            logger.exception("Skipping unreadable post %s", path)
    posts.sort(key=lambda p: (p.date or datetime.date.min, p.slug), reverse=True)
    return posts


def collect_tags(posts: list[Post]) -> list[str]:
    """Return the sorted unique tags across the given posts."""
    return sorted({t for p in posts for t in p.tags})
```

```python
"""Tests for zach_taiji_blog.posts."""

import datetime
from pathlib import Path

from zach_taiji_blog.posts import collect_tags, load_all_posts, load_post

FIXTURE = Path("fixtures/sample_post.md")


def test_load_post_metadata() -> None:
    post = load_post(FIXTURE)
    assert post.slug == "sample_post"
    assert post.title == "Sample Post"
    assert post.author == "Zach Taiji"
    assert post.date == datetime.date(2026, 7, 1)
    assert post.tags == ("python", "testing")
    assert "<mark>world</mark>" in post.html


def test_title_falls_back_to_slug(tmp_path: Path) -> None:
    (tmp_path / "my-post.md").write_text("no frontmatter here")
    assert load_post(tmp_path / "my-post.md").title == "My Post"


def test_load_all_posts_sorted(tmp_path: Path) -> None:
    (tmp_path / "old.md").write_text("---\ndate: 2020-01-01\n---\nold")
    (tmp_path / "new.md").write_text("---\ndate: 2026-01-01\n---\nnew")
    (tmp_path / "undated.md").write_text("undated")
    slugs = [p.slug for p in load_all_posts(tmp_path)]
    assert slugs == ["new", "old", "undated"]


def test_collect_tags() -> None:
    posts = [load_post(FIXTURE)]
    assert collect_tags(posts) == ["python", "testing"]
```
